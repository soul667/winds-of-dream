296vector<cv::Point2f> Wind::transform_points(vector<cv::Point2f> points_now,
297 float theta) {
298 vector<cv::Point2f> OUT;
299 for (const auto &point : points_now) {
300 // // 复平面以中心为原点，向量旋转θ公式
301 double x = point.x - circle_center.x;
302 double y = point.y - circle_center.y;
303 cv::Point2f temp = point - circle_center;
304
305 temp.x = x * cos(theta) + y * sin(theta) + circle_center.x;
306 temp.y = -x * sin(theta) + y * cos(theta) + circle_center.y;
307 OUT.push_back(temp);
308 }
309 return OUT;

318double Wind::calculate_omega(float theta, float last_theta, float timeline,
319 float lasttimeline, bool &wind_jump_flag) {
320 wind_jump_flag = false;
321 double temp_thetadot, deltatime = timeline - lasttimeline,
322 deltatheta = theta - last_theta;
323
324 if (abs(deltatheta) > 300) {
325 if (theta < ANGLE_JUMP_0_360_MIN || theta > ANGLE_JUMP_0_360_MAX) {
326 deltatheta = (deltatheta > 0) ? (deltatheta - 360) : (360 + deltatheta);
327 }
328 }
329 // 359 to 0.1 deltatheta<0 逆时针 0.1-359=-358.9 ans=360+(-358.9)
330 // 0.1 to 359 deltatheta>0 顺时针 359-0.1=358.9 ans=358.9-360
331 // 角度定义为： 以x轴正半轴为0度，逆时针转动的角度
332
333 // start wind jump judge
334 double deltatheta1 = deltatheta;
335 for (int d = 1; d <= 4; d++) {
336 if (deltatheta > 72 * (d - 1) + 65 || -deltatheta > 72 * (d - 1) + 65) {
337 deltatheta1 = (deltatheta > 0)
338 ? (deltatheta - 72 * d)
339 : (deltatheta + 72 * d); // >0就肯定是第一种
340 wind_jump_flag = true;
341 }
342 }
343
344 temp_thetadot = (wind_jump_flag) ? last_thetadot
345 : ((deltatheta1 / 180) * CV_PI) / deltatime;
346 last_thetadot = temp_thetadot;
347 return temp_thetadot;
348}
