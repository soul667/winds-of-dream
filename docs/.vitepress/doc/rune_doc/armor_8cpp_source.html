9bool Roi_rect::make_safe() {
10 cv::Size size = cv::Size(WIDTH, HEIGHT);
11 if (x < 0)
12 x = 0;
13 if (width > size.width)
14 width = size.width;
15 if (x + width > size.width)
16 x = size.width - width;
17 if (y < 0)
18 y = 0;
19 if (height > size.height)
20 height = size.height;
21 if (y + height > size.height)
22 y = size.height - height;
23 if (width <= 0 || height <= 0) {
24 x = 0;
25 y = 0;
26 width = size.width;
27 height = size.height;
28 return false;
29 }
30 return true;

244void Armor::init(const cv::Mat &camMatrix, const cv::Mat &camDist) {
245 KF.init();
246#if USE_SVM
247 svm_obj.init_model();
248#endif
249#if USE_DNN_NUMBER
250 dnn_obj.load_net_param(DNN_PARAMS_PATH);
251#endif
252 _CAM_MATRIX = camMatrix.clone();
253 _DISTORTION_COEFF = camDist.clone();
254 _solve.init(_CAM_MATRIX, _DISTORTION_COEFF);
255 tracker_ = std::make_unique<Tracker>(M_Params.MAX_DISTANCE_DIS, 1e9);
256 readParams();
257 _enemy_color = static_cast<ColorChannels>(define_params.enemy_color);
258 auto f = [this](const Eigen::VectorXd &x) {
259 Eigen::VectorXd x_new = x;
260 x_new(0) += x(1) * dt_;
261 x_new(2) += x(3) * dt_;
262 x_new(4) += x(5) * dt_;
263 x_new(6) += x(7) * dt_;
264 return x_new;
265 };
266 // J_f - Jacobian of process function
267 auto j_f = [this](const Eigen::VectorXd &) {
268 Eigen::MatrixXd f(9, 9);
269 // clang-format off
270 f << 1, dt_, 0, 0, 0, 0, 0, 0, 0,
271 0, 1, 0, 0, 0, 0, 0, 0, 0,
272 0, 0, 1, dt_, 0, 0, 0, 0, 0,
273 0, 0, 0, 1, 0, 0, 0, 0, 0,
274 0, 0, 0, 0, 1, dt_, 0, 0, 0,
275 0, 0, 0, 0, 0, 1, 0, 0, 0,
276 0, 0, 0, 0, 0, 0, 1, dt_, 0,
277 0, 0, 0, 0, 0, 0, 0, 1, 0,
278 0, 0, 0, 0, 0, 0, 0, 0, 1;
279 // clang-format on
280 return f;
281 };
282 // h - Observation function
283 auto h = [](const Eigen::VectorXd &x) {
284 Eigen::VectorXd z(4);
285 double xc = x(0), yc = x(2), yaw = x(6), r = x(8);
286 z(0) = xc - r * cos(yaw); // xa
287 z(1) = yc - r * sin(yaw); // ya
288 z(2) = x(4); // za
289 z(3) = x(6); // yaw
290 return z;
291 };
292 // J_h - Jacobian of observation function
293 auto j_h = [](const Eigen::VectorXd &x) {
294 Eigen::MatrixXd h(4, 9);
295 double yaw = x(6), r = x(8);
296 // clang-format off
297 // xc v_xc yc v_yc za v_za yaw v_yaw r
298 h << 1, 0, 0, 0, 0, 0, r*sin(yaw), 0, -cos(yaw),
299 0, 0, 1, 0, 0, 0, -r*cos(yaw),0, -sin(yaw),
300 0, 0, 0, 0, 1, 0, 0, 0, 0,
301 0, 0, 0, 0, 0, 0, 1, 0, 0;
302 // clang-format on
303 return h;
304 };
305 // update_Q - process noise covariance matrix
306 // s2qxyz_ = 20.0;
307 // s2qyaw_ = 100.0;
308 // s2qr_ = 800.0;
309 auto u_q = [this]() {
310 Eigen::MatrixXd q(9, 9);
311 double t = dt_, x = s2qxyz_, y = s2qyaw_, r = s2qr_;
312 double q_x_x = pow(t, 4) / 4 * x, q_x_vx = pow(t, 3) / 2 * x,
313 q_vx_vx = pow(t, 2) * x;
314 double q_y_y = pow(t, 4) / 4 * y, q_y_vy = pow(t, 3) / 2 * x,
315 q_vy_vy = pow(t, 2) * y;
316 double q_r = pow(t, 4) / 4 * r;
317 // clang-format off
318 // xc v_xc yc v_yc za v_za yaw v_yaw r
319 q << q_x_x, q_x_vx, 0, 0, 0, 0, 0, 0, 0,
320 q_x_vx, q_vx_vx,0, 0, 0, 0, 0, 0, 0,
321 0, 0, q_x_x, q_x_vx, 0, 0, 0, 0, 0,
322 0, 0, q_x_vx, q_vx_vx,0, 0, 0, 0, 0,
323 0, 0, 0, 0, q_x_x, q_x_vx, 0, 0, 0,
324 0, 0, 0, 0, q_x_vx, q_vx_vx,0, 0, 0,
325 0, 0, 0, 0, 0, 0, q_y_y, q_y_vy, 0,
326 0, 0, 0, 0, 0, 0, q_y_vy, q_vy_vy,0,
327 0, 0, 0, 0, 0, 0, 0, 0, q_r;
328 // clang-format on
329 return q;
330 };
331 // update_R - measurement noise covariance matrix
332
333 auto u_r = [this](const Eigen::VectorXd &z) {
334 Eigen::DiagonalMatrix<double, 4> r;
335 double x = r_xyz_factor;
336 r.diagonal() << abs(x * z[0]), abs(x * z[1]), abs(x * z[2]), r_yaw;
337 return r;
338 };
339 // P - error estimate covariance matrix
340 Eigen::DiagonalMatrix<double, 9> p0;
341 p0.setIdentity();
342 tracker_->ekf = ExtendedKalmanFilter{f, h, j_f, j_h, u_q, u_r, p0};
343}
