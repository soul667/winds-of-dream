42void Tracker::update(const Armors_ &armors) {
43 // KF predict
44 // Eigen::VectorXd ekf_prediction = ekf.predict();
45
46 matched = false;
47 // Use KF prediction as default target state if no matched armor is found
48 // target_state = ekf_prediction;
49
50 if (!armors.empty()) {
51 vector<Armor_> match_again;
52 //匹配二位距离最小，三维条件满足
53
54 for (const auto &armor : armors) {
55 Eigen::Vector3d position_vec(armor.point.x, armor.point.y, armor.point.z);
56 double position_diff = (last_point - position_vec).norm();
57 // 三维点距离最近的 二维距离已经筛过了
58 if (position_diff < max_match_distance_) {
59 // tracked_armor = armor;
60 match_again.emplace_back(armor);
61 // }
62 }
63 }
64 double angle_ = tracked_armor.rotation_angle;
65 auto priority_hit_angle_ = priority_hit_angle;
66 sort(match_again.begin(), match_again.end(),
67 [angle_, priority_hit_angle_](const Armor_ &a, const Armor_ &b) {
68 double angle_a = abs(a.rotation_angle - angle_);
69 double angle_b = abs(b.rotation_angle - angle_);
70
71 if (a.rotation_angle >= priority_hit_angle_ &&
72 a.rotation_angle <= 90 + priority_hit_angle_)
73 return 1;
74 if (angle_a < angle_b)
75 return 1;
76 return 0;
77 });
78 if (match_again.size() != 0) {
79 tracked_armor = match_again[0];
80 matched = true;
81 auto p = tracked_armor.point;
82 last_point = Eigen::Vector3d(p.x, p.y, p.z);
83 measurement = Eigen::Vector4d(p.x, p.y, p.z, 0);
84
85 } else {
86 // handleArmorJump(same_id_armor);
87 }
88 }
89
90 // Tracking state machine
91 if (tracker_state == DETECTING) {
92 if (matched) {
93 detect_count_++;
94 // tracking_thres
95 if (detect_count_ > 1) {
96 detect_count_ = 0;
97 tracker_state = TRACKING;
98 }
99 } else {
100 detect_count_ = 0;
101 tracker_state = LOST;
102 }
103 } else if (tracker_state == TRACKING) {
104 if (!matched) {
105 tracker_state = TEMP_LOST;
106 lost_count_++;
107 tracker_count_ = 0;
108
109 } else {
110 tracker_count_++;
111 }
112 } else if (tracker_state == TEMP_LOST) {
113 if (!matched) {
114 lost_count_++; // lost_thres
115 if (lost_count_ > lost_thres) {
116 lost_count_ = 0;
117 tracker_state = LOST;
118 }
119 } else {
120 tracker_state = TRACKING;
121 lost_count_ = 0;
122 }
123 }

126void Tracker::initEKF(const Armor_ &a) {
127 double xa = a.point.x;
128 double ya = a.point.y;
129 double za = a.point.z;
130 last_yaw_ = 0;
131 // double yaw = orientationToYaw(a.pose.orientation);
132
133 // Set initial position at 0.2m behind the target;
134 // delete the yaw
135 target_state = Eigen::VectorXd::Zero(9);
136 double r = 0.26;
137 double xc = xa + r * cos(0); // cos(yaw)
138 double yc = ya + r * sin(0); // sin(yaw)
139 dz = 0, another_r = r;
140 target_state << xc, 0, yc, 0, za, 0, 0, 0, r;
141
142 ekf.setState(target_state);
143}

156void Tracker::handleArmorJump(const Armor_ &armor) {
157 // double yaw = orientationToYaw(current_armor.pose.orientation);
158 target_state(6) = 0;
159 updateArmorsNum(current_armor);
160 // Only 4 armors has 2 radius and height
161 if (tracked_armors_num == ArmorsNum::NORMAL_4) {
162 dz = target_state(4) - current_armor.point.z;
163 target_state(4) = current_armor.point.z;
164 std::swap(target_state(8), another_r);
165 }
166 // If position difference is larger than max_match_distance_,
167 // take this case as the ekf diverged, reset the state
168 auto p = current_armor.point;
169 Eigen::Vector3d current_p(p.x, p.y, p.z);
170 Eigen::Vector3d infer_p = getArmorPositionFromState(target_state);
171 if ((current_p - infer_p).norm() > max_match_distance_) {
172 double r = target_state(8);
173 target_state(0) = p.x + r * cos(0); // xc yaw
174 target_state(1) = 0; // vxc
175 target_state(2) = p.y + r * sin(0); // yc
176 target_state(3) = 0; // vyc
177 target_state(4) = p.z; // za
178 target_state(5) = 0; // vza
179 }
180
181 ekf.setState(target_state);
182}

196Eigen::Vector3d Tracker::getArmorPositionFromState(const Eigen::VectorXd &x) {
197 // Calculate predicted position of the current armor
198 double xc = x(0), yc = x(2), za = x(4);
199 double yaw = x(6), r = x(8);
200 double xa = xc - r * cos(yaw);
201 double ya = yc - r * sin(yaw);
202 return Eigen::Vector3d(xa, ya, za);
203}
