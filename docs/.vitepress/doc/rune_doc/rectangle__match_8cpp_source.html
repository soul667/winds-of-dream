6bool containLight(const RotatedRect &light_1_, const RotatedRect &light_2_,
7 const std::vector<RotatedRect> &lights_) {
8 Light light_1 = Light(light_1_);
9 Light light_2 = Light(light_2_);
10 std::vector<Light> lights;
11 for (const auto &light : lights_) {
12 lights.emplace_back(Light(light));
13 }
14
15 auto points = std::vector<cv::Point2f>{light_1.top, light_1.bottom,
16 light_2.top, light_2.bottom};
17 auto bounding_rect = cv::boundingRect(points);
18
19 for (const auto &test_light : lights) {
20 if (test_light.center == light_1.center ||
21 test_light.center == light_2.center)
22 continue;
23
24 if (bounding_rect.contains(test_light.top) ||
25 bounding_rect.contains(test_light.bottom) ||
26 bounding_rect.contains(test_light.center)) {
27 return true;
28 }
29 }
30
31 return false;
32}
