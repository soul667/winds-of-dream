67void WindEKF::correct(float timeline, float theta, int direction) {
68 // std::cout<<direction<<"direc"<<std::endl;
69 // std::cout<<"timeline:::::"<<timeline<<std::endl;
70 // std::cout<<theta/CV_PI<<" "<<std::endl;
71 float real = x_hat.at<float>(0), imag = x_hat.at<float>(1),
72 thetadot = x_hat.at<float>(2), time = x_hat.at<float>(3);
73 float ampEst = x_hat.at<float>(4), omegaEst = x_hat.at<float>(5);
74 float dt = timeline - Lasttimestamp;
75 // std::cout<<"dt"<<dt<<std::endl;
76 f_x.at<float>(0) = real * cos(thetadot * dt) - imag * sin(thetadot * dt);
77 f_x.at<float>(1) = real * sin(thetadot * dt) + imag * cos(thetadot * dt);
78 f_x.at<float>(2) =
79 (ampEst * sin(omegaEst * (time + dt)) + 2.09 - ampEst) * direction;
80 f_x.at<float>(3) = time + dt;
81 f_x.at<float>(4) = ampEst;
82 f_x.at<float>(5) = omegaEst;
83
84 F.at<float>(0, 0) = cos(thetadot * dt);
85 F.at<float>(0, 1) = -sin(thetadot * dt);
86 F.at<float>(0, 2) =
87 -real * dt * sin(thetadot * dt) - imag * dt * cos(thetadot * dt);
88 F.at<float>(1, 0) = sin(thetadot * dt);
89 F.at<float>(1, 1) = -cos(thetadot * dt);
90 F.at<float>(1, 2) =
91 real * dt * cos(thetadot * dt) - imag * dt * sin(thetadot * dt);
92 F.at<float>(2, 3) = (omegaEst * ampEst * cos(omegaEst * time)) * direction;
93 F.at<float>(2, 4) = (sin(omegaEst * time) - 1) * direction;
94 F.at<float>(2, 5) = (time * ampEst * cos(omegaEst * time)) *
95 direction; // (3,3),(4,4),(5,5)为1,F[6x6]
96 P_minus = F * P * F.t() + Q; //计算先验协方差
97 H.at<float>(0) = -imag / (real * real + imag * imag);
98 H.at<float>(1) = real / (real * real + imag * imag);
99 float temp_theta = atan2(f_x.at<float>(1), f_x.at<float>(0));
100 if (temp_theta < 0) {
101 temp_theta += CV_PI * 2;
102 }
103 est_theta = temp_theta;
104 // std::cout<<"temp_theta"<<temp_theta<<std::endl;
105 float residual = theta - temp_theta;
106 // std::cout<<"residual"<<residual<<std::endl;
107 cv::Mat temp = H * P_minus * H.t() + R;
108 chiSquare = (residual * (1 / temp.at<float>(0)) * residual);
109 // std::cout<<"chiSquare"<<chiSquare<<std::endl;
110 if (chiSquare < p.chiSquare_MAX) {
111 K = P_minus * H.t() / temp.at<float>(0); //计算卡尔曼增益
112 x_hat = f_x + K * residual;
113 cv::Mat unit_matrix;
114 unit_matrix =
115 cv::Mat::eye(cv::Size(p.dynamParams, p.dynamParams), CV_32FC1);
116 P = (unit_matrix - K * H) * P_minus; //更新协方差误差
117 } else {
118 x_hat = f_x;
119 x_hat.at<float>(0) = cos(theta);
120 x_hat.at<float>(1) = sin(theta);
121 P = P_minus;
122 }
123 amp = x_hat.at<float>(4);
124 omega = x_hat.at<float>(5);
125 theta_dot = x_hat.at<float>(2);
126 phi = (x_hat.at<float>(3) - timeline) * omega;
127 Lasttimestamp = timeline;
128 Last_theta = theta;
129}

140float WindEKF::predict(float flytime, float timeline) { // s
141 float theta = 0.0;
142 float delta_time = flytime / 50000;
143 for (float i = 0; i <= flytime; i += delta_time) {
144 float spd = amp * sin(omega * (timeline + i) + phi) + 2.09 - amp;
