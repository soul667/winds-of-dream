84cv::Point3f Solve::get_real_Point3f(double pitch, double yaw) {
85 // pitch=3.1415926*pitch/180;
86 // yaw=3.1415926*yaw/180;
87
88 // ToOpenCV4.2.0
89 double k[3][3] = {1, 0, 0, 0, 0, 1, 0, -1, 0};
90 cv::Mat K(3, 3, CV_64FC1 /*ToOpenCV4.2.0*/, k);
91
92 // ToOpenCV4.2.0
93 double offset[3] = {_offset_x, _offset_y, _offset_z};
94 cv::Mat OFFSET(3, 1, CV_64FC1 /*ToOpenCV4.2.0*/, offset);
95
96 // ToOpenCV4.2.0
97 double xyz[3] = {_tVec.at<double>(0, 0), _tVec.at<double>(1, 0),
98 _tVec.at<double>(2, 0)};
99 cv::Mat XYZ(3, 1, CV_64FC1, xyz);
100
101 // ToOpenCV4.2.0
102 double z_rotate[3][3] = {cos(yaw), -sin(yaw), 0, sin(yaw), cos(yaw),
103 0, 0, 0, 1};
104 cv::Mat Z_ROTATE(3, 3, CV_64FC1 /*ToOpenCV4.2.0*/, z_rotate);
105
106 // ToOpenCV4.2.0
107 double x_rotate[3][3] = {1, 0, 0, 0, cos(pitch),
108 -sin(pitch), 0, sin(pitch), cos(pitch)};
109 cv::Mat X_ROTATE(3, 3, CV_64FC1 /*ToOpenCV4.2.0*/, x_rotate);
110
111 cv::Mat result = Z_ROTATE * X_ROTATE * (K * XYZ + OFFSET);
112 // readParams();
113 return cv::Point3f(result.at<double>(0, 0), result.at<double>(1, 0),
114 result.at<double>(2, 0));
115}

117cv::Point2f Solve::get_Point2f(cv::Point3f input, float pitch, float yaw) {
118 cv::Mat p(3, 1, CV_64F, cv::Scalar(1));
119 p.at<double>(0, 0) = input.x;
120 p.at<double>(1, 0) = input.y;
121 p.at<double>(2, 0) = input.z;
122
123 // ToOpenCV4.2.0
124 double k[3][3] = {1, 0, 0, 0, 0, 1, 0, -1, 0};
125 cv::Mat K(3, 3, CV_64F /*ToOpenCV4.2.0*/, k);
126
127 // ToOpenCV4.2.0
128 double offset[3] = {_offset_x, _offset_y, _offset_z};
129 cv::Mat OFFSET(3, 1, CV_64F, offset);
130
131 // ToOpenCV4.2.0
132 double zchange[3][3] = {cos(yaw), -sin(yaw), 0, sin(yaw), cos(yaw),
133 0, 0, 0, 1};
134 cv::Mat zc(3, 3, CV_64F /*ToOpenCV4.2.0*/, zchange);
135
136 // ToOpenCV4.2.0
137 double xchange[3][3] = {1, 0, 0, 0, cos(pitch),
138 -sin(pitch), 0, sin(pitch), cos(pitch)};
139 cv::Mat xc(3, 3, CV_64F, xchange);
140
141 cv::Mat _Point;
142 _Point = K.inv() * (xc.inv() * zc.inv() * p - OFFSET);
143
144 cv::Mat Point(3, 1, CV_64F, cv::Scalar(1));
145 Point.at<double>(0, 0) = _Point.at<double>(0, 0);
146 Point.at<double>(1, 0) = _Point.at<double>(1, 0);
147 Point.at<double>(2, 0) = _Point.at<double>(2, 0);
148 cv::Mat output = _CAM_MATRIX * Point;
149
150 output.at<double>(0, 0) /= output.at<double>(2, 0);
151 output.at<double>(1, 0) /= output.at<double>(2, 0);
152 output.at<double>(2, 0) /= output.at<double>(2, 0);
153
154 return cv::Point2f(output.at<double>(0, 0), output.at<double>(1, 0));
155}
