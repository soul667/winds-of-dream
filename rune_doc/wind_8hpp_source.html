164 void check_again(vector<Moon_use> moon_points) {
165 for (auto &i : moon_points) {
166 // if (i.hu0 < 1 || i.hu1 < 1)
167 // continue;
168 double dist = cv::norm(i.center - center);
169 if (dist > max_dist) {
170 max_dist = dist;
171 }
172 if (dist < min_dist) {
173 min_dist = dist;
174 }
175
176 cv::Point2f p[4];
177 i.points(p);
178 double max_ = 0, min_ = 1e4;
179 for (auto &point : p) {
180 cv::Point2f use = point - i.center;
181 double dist_ =
182 (cv::norm((use).dot(center - i.center)) / cv::norm(use));
183 if (dist > max_) {
184 max_ = dist_;
185 }
186 if (dist < min_) {
187 min_ = dist_;
188 }
189 }
190 min_dis_symmetry_axis +=
191 abs((max_ - min_) / cv::norm(center - i.center));
192 }
193 ratio_dist = min_dist / max_dist;
194 min_dis_symmetry_axis /= moon_points.size();
195 }
