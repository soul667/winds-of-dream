9double gradient_descent::LossFunction(estimate_params &p1,
10 std::deque<measure_params> &p2) {
11 double err = 0, F = 0;
12 double t_phi = p1.phi * 2 * CV_PI;
13 double t_amp = p1.amp * 0.265 + 0.78;
14 double t_omega = p1.omega * 0.116 + 1.884;
15 std::deque<measure_params>::iterator it2 = p2.begin();
16 int m = 0;
17 do {
18 err =
19 (*it2).thetadot -
20 direction * (t_amp * sin((double)t_omega * ((*it2).timeline) + t_phi) +
21 2.09 - t_amp);
22 F = F + err * err;
23 it2++;
24 m++;
25 } while (it2 != p2.end() - 1);
26 return F / (double)m;

28void gradient_descent::start_train() {
29 is_training = true;
30 train_params.assign(m_params.begin(), m_params.end());
31 int k = 0;
32 lossval = LossFunction(e, train_params);
33 momentum.phi = 0;
34 momentum.amp = 0;
35 momentum.omega = 0;
36 do {
37 estimate_params last_params = {e.phi, e.amp, e.omega};
38 lastlossval = lossval;
39 double dphi = 0.0001, damp = 0.0001, domega = 0.0001;
40 estimate_params tempparams = {e.phi, e.amp, e.omega};
41 tempparams.phi = e.phi + dphi;
42 double dfdphi = (LossFunction(tempparams, train_params) - lossval) / dphi;
43 tempparams.phi = e.phi;
44 tempparams.amp = e.amp + damp;
45 double dfdamp = (LossFunction(tempparams, train_params) - lossval) / damp;
46 tempparams.amp = e.amp;
47 tempparams.omega = e.omega + domega;
48 double dfdomega =
49 (LossFunction(tempparams, train_params) - lossval) / domega;
50 if (k == 0) {
51 sigma[0] = dfdphi;
52 sigma[1] = dfdamp;
53 sigma[2] = dfdomega;
54 } else {
55 sigma[0] = sqrt(p.alpha * lastsigma[0] * lastsigma[0] +
56 (1 - p.alpha) * dfdphi * dfdphi);
57 sigma[1] = sqrt(p.alpha * lastsigma[1] * lastsigma[1] +
58 (1 - p.alpha) * dfdamp * dfdamp);
59 sigma[2] = sqrt(p.alpha * lastsigma[2] * lastsigma[2] +
60 (1 - p.alpha) * dfdomega * dfdomega);
61 }
62 estimate_params deltaparams{-dfdphi * p.learning_rate / sigma[0],
63 -dfdamp * p.learning_rate / sigma[1],
64 -dfdomega * p.learning_rate / sigma[2]};
65 estimate_params fixeddeltaparams = momentum * p.lambda + deltaparams;
66 e = e + fixeddeltaparams;
67 // limit_value(e.amp, p.ampMax, 1);
68 // limit_value(e.amp, p.ampMin, 0);
69 // limit_value(e.omega, p.omegaMax, 1);
70 // limit_value(e.omega, p.omegaMin, 0);
71
72 lossval = LossFunction(e, train_params);
73 // std::cout <<"lossval"<<lossval<<std::endl;
74 deltalossval = lossval - lastlossval;
75 lastsigma[0] = sigma[0];
76 lastsigma[1] = sigma[1];
77 lastsigma[2] = sigma[2];
78 momentum = fixeddeltaparams;
79 k += 1;
80 } while (k < p.train_max_times && fabs(deltalossval) > p.min_deltalossval);
81 train_flag = false;
82
83 result.phi = e.phi * 2 * CV_PI;
84 result.amp = e.amp * 0.265 + 0.78;
85 result.omega = e.omega * 0.116 + 1.884;
86 train_params.clear();
87 is_training = false;
88 p.learning_rate = 0.01;
89}
