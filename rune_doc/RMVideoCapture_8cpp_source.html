122bool RMVideoCapture::setVideoFormat(size_t width, size_t height, bool mjpg)
123{
124 if (_width == width && _height == height)
125 {
126 return true;
127 }
128 _width = width;
129 _height = height;
130 _frameCount = 0;
131 struct v4l2_format fmt;
132 memset(&fmt, 0, sizeof(fmt));
133 fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
134 fmt.fmt.pix.width = width;
135 fmt.fmt.pix.height = height;
136 if (mjpg == true)
137 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_MJPEG;
138 else
139 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV;
140 fmt.fmt.pix.field = V4L2_FIELD_ANY;
141
142 if (-1 == xioctl(_camera_fd, VIDIOC_S_FMT, &fmt))
143 {
144 cout << "Setting Pixel Format failed" <<endl;
145 return false;
146 }
147 return true;
148}

239bool RMVideoCapture::initMMap()
240{
241 struct v4l2_requestbuffers bufrequest;
242 bufrequest.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
243 bufrequest.memory = V4L2_MEMORY_MMAP;
244 bufrequest.count = _buffer_size;
245
246 if(ioctl(_camera_fd, VIDIOC_REQBUFS, &bufrequest) < 0)
247 {
248 printf("Requesting buffer failed.\n");
249 return false;
250 }
251
252 for(size_t i = 0; i < _buffer_size; ++i)
253 {
254 struct v4l2_buffer bufferinfo;
255 memset(&bufferinfo, 0, sizeof(bufferinfo));
256 bufferinfo.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
257 bufferinfo.memory = V4L2_MEMORY_MMAP;
258 bufferinfo.index = i; /* Queueing buffer index 0. */
259
260 // mmap for buffers
261 if(ioctl(_camera_fd, VIDIOC_QUERYBUF, &bufferinfo) < 0)
262 {
263 perror("Buffer map failed.\n");
264 return false;
265 }
266
267 _buffers[i].ptr = mmap(
268 NULL,
269 bufferinfo.length,
270 PROT_READ | PROT_WRITE,
271 MAP_SHARED,
272 _camera_fd,
273 bufferinfo.m.offset);
274 _buffers[i].length = bufferinfo.length;
275
276 if(_buffers[i].ptr == MAP_FAILED)
277 {
278 perror("Buffer map failed.\n");
279 return false;
280 }
281
282 memset(_buffers[i].ptr, 0, bufferinfo.length);
283
284 // Put the buffer in the incoming queue.
285 memset(&bufferinfo, 0, sizeof(bufferinfo));
286 bufferinfo.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
287 bufferinfo.memory = V4L2_MEMORY_MMAP;
288 bufferinfo.index = i;
289 if(ioctl(_camera_fd, VIDIOC_QBUF, &bufferinfo) < 0)
290 {
291 perror("Querying buffer ");
292 return false;
293 }
294 }
295
296 return true;
297}

299bool RMVideoCapture::startStream()
300{
301 _frameCount = 0;
302 refreshVideoFormat();
303 if(initMMap() == false) return false;
304
305 __u32 type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
306 if(ioctl(_camera_fd, VIDIOC_STREAMON, &type) < 0)
307 {
308 perror("Starting stream failed.\n");
309 return false;
310 }
311 return true;

314bool RMVideoCapture::closeStream()
315{
316 _frameCount = 0;
317 _bufferIdx = 0;
318 __u32 type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
319 if(ioctl(_camera_fd, VIDIOC_STREAMOFF, &type) < 0)
320 {
321 perror("VIDIOC_STREAMOFF");
322 return false;
323 }
324 for(size_t i = 0; i < _buffer_size; ++i)
325 {
326 munmap(_buffers[i].ptr, _buffers[i].length);
327 }
328 return true;
329}

341void RMVideoCapture::cvtRaw2Mat(void* data, cv::Mat& image)
342{
343 if (_format == V4L2_PIX_FMT_MJPEG)
344 {
345 //OpenCV method, 60fps
346 cv::Mat mjpg(_height, _width, CV_8UC3, data);
347 image = cv::imdecode(mjpg, CV_LOAD_IMAGE_COLOR);
348
349 //libjpeg-turbo method, 30fps
350 //image = Jpeg2Mat((uint8_t*)data,length);
351 }
352 else if(_format == V4L2_PIX_FMT_YUYV)
353 {
354 cv::Mat yuyv(_height, _width, CV_8UC2, data);
355 cv::cvtColor(yuyv, image, CV_YUV2BGR_YUYV);
356 }

454void RMVideoCapture::info()
455{
456 struct v4l2_capability caps = {};
457 if (-1 == xioctl(_camera_fd, VIDIOC_QUERYCAP, &caps))
458 {
459 perror("Querying Capabilities\n");
460 return;
461 }
462
463 printf( "Driver Caps:\n"
464 " Driver: \"%s\"\n"
465 " Card: \"%s\"\n"
466 " Bus: \"%s\"\n"
467 " Version: %d.%d\n"
468 " Capabilities: %08x\n",
469 caps.driver,
470 caps.card,
471 caps.bus_info,
472 (caps.version>>16)&&0xff,
473 (caps.version>>24)&&0xff,
474 caps.capabilities);
475
476
477 struct v4l2_cropcap cropcap;
478 memset(&cropcap, 0, sizeof(cropcap));
479 cropcap.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
480 if (-1 == xioctl (_camera_fd, VIDIOC_CROPCAP, &cropcap))
481 {
482 perror("Querying Cropping Capabilities\n");
483 return;
484 }
485
486 printf( "Camera Cropping:\n"
487 " Bounds: %dx%d+%d+%d\n"
488 " Default: %dx%d+%d+%d\n"
489 " Aspect: %d/%d\n",
490 cropcap.bounds.width, cropcap.bounds.height, cropcap.bounds.left, cropcap.bounds.top,
491 cropcap.defrect.width, cropcap.defrect.height, cropcap.defrect.left, cropcap.defrect.top,
492 cropcap.pixelaspect.numerator, cropcap.pixelaspect.denominator);
493
494 struct v4l2_fmtdesc fmtdesc;
495 memset(&fmtdesc, 0, sizeof(fmtdesc));
496 fmtdesc.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
497 char fourcc[5] = {0};
498 char c, e;
499 printf(" FMT : CE Desc\n--------------------\n");
500 while (0 == xioctl(_camera_fd, VIDIOC_ENUM_FMT, &fmtdesc))
501 {
502 strncpy(fourcc, (char *)&fmtdesc.pixelformat, 4);
503 c = fmtdesc.flags & 1? 'C' : ' ';
504 e = fmtdesc.flags & 2? 'E' : ' ';
505 printf(" %s: %c%c %s\n", fourcc, c, e, fmtdesc.description);
506 fmtdesc.index++;
507 }
508
509 struct v4l2_format fmt;
510 memset(&fmt, 0, sizeof(fmt));
511 fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
512 if (-1 == xioctl(_camera_fd, VIDIOC_G_FMT, &fmt))
513 {
514 perror("Querying Pixel Format\n");
515 return;
516 }
517 strncpy(fourcc, (char *)&fmt.fmt.pix.pixelformat, 4);
518 printf( "Selected Camera Mode:\n"
519 " Width: %d\n"
520 " Height: %d\n"
521 " PixFmt: %s\n"
522 " Field: %d\n",
523 fmt.fmt.pix.width,
524 fmt.fmt.pix.height,
525 fourcc,
526 fmt.fmt.pix.field);
527
528 struct v4l2_streamparm streamparm;
529 memset(&streamparm, 0, sizeof(streamparm));
530 streamparm.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
531 if (-1 == ioctl(_camera_fd, VIDIOC_G_PARM, &streamparm))
532 {
533 perror("Querying Frame Rate\n");
534 return;
535 }
536 printf( "Frame Rate: %f\n====================\n",
537 (float)streamparm.parm.capture.timeperframe.denominator /
538 (float)streamparm.parm.capture.timeperframe.numerator);
539}
