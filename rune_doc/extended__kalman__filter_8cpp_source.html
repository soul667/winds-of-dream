16Eigen::MatrixXd ExtendedKalmanFilter::predict() {
17 F = jacobian_f(x_post), Q = update_Q();
18
19 x_pri = f(x_post);
20 P_pri = F * P_post * F.transpose() + Q;
21
22 // handle the case when there will be no measurement before the next predict
23 x_post = x_pri;
24 P_post = P_pri;
25
26 return x_pri;
27}

29Eigen::MatrixXd ExtendedKalmanFilter::update(const Eigen::VectorXd &z) {
30 H = jacobian_h(x_pri), R = update_R(z);
31
32 K = P_pri * H.transpose() * (H * P_pri * H.transpose() + R).inverse();
33 x_post = x_pri + K * (z - h(x_pri));
34 P_post = (I - K * H) * P_pri;
35
36 return x_post;
37}
