76 int Serial::openPort()
77 {
78 _serialFd = open("/dev/ttyUSB_0", O_RDWR | O_NOCTTY | O_NONBLOCK);
79 if (_serialFd == -1)
80 {
81 cout << "Open serial port failed." << endl;
82 // LOG_FIRST_N(INFO, 1)<<"Open serial port failed." ;
83 return _errorCode = SYSTEM_ERROR;
84 ;
85 }
86
87 termios tOption; // 串口配置结构体
88 tcgetattr(_serialFd, &tOption); // 获取当前设置
89 cfmakeraw(&tOption);
90 cfsetispeed(&tOption, B460800); // 接收波特率
91 cfsetospeed(&tOption, B460800); // 发送波特率
92 tcsetattr(_serialFd, TCSANOW, &tOption);
93 tOption.c_cflag &= ~PARENB;
94 tOption.c_cflag &= ~CSTOPB;
95 tOption.c_cflag &= ~CSIZE;
96 tOption.c_cflag |= CS8;
97 tOption.c_cflag &= ~INPCK;
98 tOption.c_cflag |= (B460800 | CLOCAL | CREAD); // 设置波特率，本地连接，接收使能
99 tOption.c_cflag &= ~(INLCR | ICRNL);
100 tOption.c_cflag &= ~(IXON);
101 tOption.c_lflag &= ~(ICANON | ECHO | ECHOE | ISIG);
102 tOption.c_oflag &= ~OPOST;
103 tOption.c_oflag &= ~(ONLCR | OCRNL);
104 tOption.c_iflag &= ~(ICRNL | INLCR);
105 tOption.c_iflag &= ~(IXON | IXOFF | IXANY);
106 tOption.c_cc[VTIME] = 1; // 只有设置为阻塞时这两个参数才有效
107 tOption.c_cc[VMIN] = 1;
108 tcflush(_serialFd, TCIOFLUSH); // TCIOFLUSH刷新输入、输出队列。
109
110 cout << "Serial preparation complete." << endl;
111 SerialOpened = 1;
112 return _errorCode = SUCCESS;
113 }

346 int Serial::receive()
347 {
348 memset(&_FeedBackData, 0, sizeof(FeedBackFrame)); // clean
349
350 int readCount = 0;
351 const auto t1 = std::chrono::high_resolution_clock::now();
352 while (readCount < int(sizeof(FeedBackFrame)))
353 {
354
355 auto t2 = std::chrono::high_resolution_clock::now();
356
357 if ((chrono::duration_cast<std::chrono::milliseconds>(t2 - t1)).count() > 10) // Time limit is 10ms
358 {
359 if (_en_debug)
360 {
361 cout << "\tReceiving time out. " << sizeof(FeedBackFrame) - readCount
362 << " bytes not received." << endl;
363 // LOG_FIRST_N(INFO, 1)<<"Receiving time out bytes not received ";
364 }
365 return _errorCode = TIME_OUT;
366 }
367
368 int onceReadCount;
369 try
370 {
371 onceReadCount = read(_serialFd, ((unsigned char *)(&_FeedBackData)) + readCount, sizeof(FeedBackFrame) - readCount);
372 }
373 catch (exception e)
374 {
375 cout << e.what() << endl;
376 return _errorCode = SYSTEM_ERROR;
377 }
378
379 if (onceReadCount == -1)
380 {
381 if (errno == EAGAIN)
382 {
383 continue;
384 }
385
386 if (_en_debug)
387 {
388 cout << "\tRead data from serial failed." << endl;
389 // LOG_FIRST_N(INFO, 1)<<"Read data from serial failed.";
390 }
391 return _errorCode = READ_WRITE_ERROR;
392 }
393
394 readCount += onceReadCount;
395 }
396
397 tcflush(_serialFd, TCIFLUSH);
398
399 if (_FeedBackData.SOF != CommSOF || _FeedBackData.EOF != CommEOF)
400 {
401 if (_en_debug)
402 {
403 cout << "\tFeed back frame SOF or EOF is not correct. SOF: " << (int)_FeedBackData.SOF << " ,EOF: " << (int)_FeedBackData.EOF << endl;
404 // LOG_FIRST_N(INFO, 1)<<"Feed back frame SOF or EOF is not correct. SOF:";
405 }
406 return _errorCode = CORRUPTED_FRAME;
407 }
408 else
409 {
410 if (_en_debug)
411 {
412 cout << "\tSerial receiving succeeded. " << endl;
413 }
414 return _errorCode = SUCCESS;
415 }
416 }
