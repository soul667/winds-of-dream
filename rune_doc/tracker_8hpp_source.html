19// 维护一个固定长度的最大队列 [O(1)]
20class MaxQueue {
21 queue<int> que;
22 deque<int> deq;
23
24public:
25 MaxQueue() {}
26 int pop_front() {
27 if (que.empty())
28 return -1;
29 int val = que.front();
30 if (val == deq.front())
31 deq.pop_front();
32 que.pop();
33 return val;
34 }
35 int max_value() {
36 // return deq.empty() ? -1 : deq.front();
37 return deq.front();
38 }
39 int size() { return que.size(); }
40 void clear() {
41 while (!que.empty())
42 que.pop();
43 while (!deq.empty())
44 deq.pop_front();
45 }
46 void emplace_back(int value) {
47 if (que.size() >= maximum_queue_size)
48 pop_front();
49
50 que.push(value);
51 while (!deq.empty() && deq.back() < value)
52 deq.pop_back();
53 deq.emplace_back(value);
54 }
55
56private:
57 int maximum_queue_size = 10;
58};

73struct Armor_ {
74 Armor_() = default;
75 explicit Armor_(std::vector<cv::Point_<float>> rect_, cv::Point3f point_,
76 string number_)
77 : rect(rect_), point(point_), number(number_) {
78 // get the bottom，上面两个点的中点
79 sort(rect.begin(), rect.end(),
80 [](cv::Point_<float> &a, cv::Point_<float> &b) { return a.y < b.y; });
81 above = (rect[0] + rect[1]) / 2;
82 bottom = (rect[2] + rect[3]) / 2;
83 center = (above + bottom) / 2;
84 // get the angle
85 double angle_ = get_angle(above - bottom);
86 rotation_angle =
87 180 -
88 acos(tan(75 * M_PI / 180) / tan(angle_ * M_PI / 180)) * 180 / 3.14;
89 width = abs(rect[0].x - rect[1].x);
90 height = abs(rect[0].y - rect[3].y);
91 lw_ratio = (width / height) *
92 (1 / (cos((rotation_angle) / 180 * 3.1415 - 3.1415 / 2)));
93 }
94
95 ~Armor_(){};
96 cv::Point3f point;
97 cv::Point2f bottom, above, center;
98 double width, height;
99 double angle = 0;
100 double lw_ratio = 0;
101 string number;
102 double rotation_angle = 0; //绕中心旋转的角度
103 std::vector<cv::Point_<float>> rect;
104 double get_angle(cv::Point2f point_1) {
105 double x = point_1.x, y = -point_1.y;
106 double angle = std::atan2(y, x) * 180 / M_PI;
107 if (x >= 0 && y <= 0) {
108 angle = angle + 360;
109 } else if ((x <= 0 && y <= 0)) {
110 angle += 360;
111 }
112 return angle;
113 }
