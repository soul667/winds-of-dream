113 /*
114 * @brief 两个全连接层之间的前向传播函数
115 * @param input 输入数组指针
116 * @param input_long 输入数组的长度
117 * @param fc_k fc层的权重矩阵数组指针
118 * @param fc_b fc层的偏置矩阵数组指针
119 * @param output 输出层数组指针
120 * @param fc_num fc层神经元个数
121 * @param flag 是否需要使用tanh激活函数
122 */
123 void DNN::Forward_propagation(const double *input, int input_long, const double *fc_k, const double *fc_b, double *output, int fc_num, bool flag)
124 {
125 memset(output, 0, sizeof(double) * fc_num);
126 // i:输出数组循环变量
127 for (int i = 0; i < fc_num; i++)
128 // j:输入数组循环变量
129 for (int j = 0; j < input_long; j++)
130 // 输出的第i个元素 += 输入图像的第j个元素值 * 权值矩阵的第j * fc_num + i个元素值
131 // j * fc_num是代表对于第j个输入元素，其前面的元素的权重的数量，需要加上这个值越过，这样加上i即这个元素对应的权重
132 *(output + i) += (*(input + j)) * (*(fc_k + i * input_long + j));
133
134 // 别忘了加上偏置项
135 for (int i = 0; i < fc_num; i++)
136 {
137 *(output + i) += *(fc_b + i);
138 if (flag == 1)
139 *(output + i) = (exp(*(output + i)) - exp(-*(output + i))) / (exp(*(output + i)) + exp(-*(output + i)));
140 }
141 return;
142 }

150 void DNN::softmax(const double *input, double *output, int num)
151 {

153 for (int i = 0; i < num; i++)
154 {
155 *(output + i) = exp(*(input + i));
156 my_sum += *(output + i);
