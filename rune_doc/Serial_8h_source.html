134 /*
135 * @Brief: 控制机器人
136 * @Input: time_duration: try for how long to send control data
137 */
138 template <typename _Rep, typename _Period>
139 int tryControl(const ControlFrame &controlData,
140 const std::chrono::duration<_Rep, _Period> &time_duration) {
141 // std::unique_lock<std::mutex> lockGuard(_mutex1/*,
142 // time_duration*/);
143 std::unique_lock<std::timed_mutex> lockGuard(_mutex, time_duration);
144 if (lockGuard.owns_lock()) {
145 control(controlData);
146 // std::this_thread::sleep_for(std::chrono::milliseconds(1));
147 return _errorCode;
148 } else {
149 if (_en_debug)
150 std::cout << "PORT_OCCUPIED" << std::endl;
151 return PORT_OCCUPIED;
152 }
153 }

161 /*
162 * @Brief: 从机器人接收信息
163 * @Input: time_duration: try for how long to receive feedback data
164 */
165 template <typename _Rep, typename _Period>
166 int tryFeedBack(FeedBackFrame &feedBackData,
167 const std::chrono::duration<_Rep, _Period> &time_duration) {
168
169 // std::unique_lock<std::mutex> lockGuard(_mutex1/*,
170 // time_duration*/);
171 std::unique_lock<std::timed_mutex> lockGuard(_mutex, time_duration);
172 if (lockGuard.owns_lock()) {
173 feedBack(feedBackData);
174 // std::this_thread::sleep_for(std::chrono::milliseconds(1));
175 return _errorCode;
176 } else {
177 return PORT_OCCUPIED;
178 }
179 }
